% Printed books have indexes
\ifxetex
	\makeindex
\fi

\begin{document}
% use chapter boxes only for printed books: uncomment \input{lib/chapterbox} and comment out \newcommand{\chapterbox}[1]{} to produce nicer looking chapter pages (but also some warnings due to older packages). 
\ifxetex
	%\input{lib/chapterbox}
	\newcommand{\chapterbox}[1]{}
\else
	\newcommand{\chapterbox}[1]{}
\fi
% ---------- Front matter
\frontmatter

% front matter chapter entries use roman page numbering (i, ii, iii, iv, ...)
\pagenumbering{roman}

% switch to basic chapter design
\input{lib/chapterfont}

% the additional title with the cover is not needed for ebooks
\ifxetex
	\input{front/title}\newpage
\fi

\input{front/foreword.tex}
\input{lib/tableofcontents}\newpage

% ---------- Main matter
\mainmatter

% reset to normal page numbering (1, 2, 3, ...)
\pagenumbering{arabic}
    
% switch back to basic chapter design
\input{lib/chapterfont}
\input{main/firstchapter.tex}
\input{main/secondchapter.tex}

\input{back/bibliography}

\end{document}
